[docs]def hex2bin(message):
    b = []
    for i in range(len(message) - 1):
        b.append(int(
            '0b' + format(int(chr(message[i]), 16), 'b').zfill(4) + format(int(chr(message[i + 1]), 16), 'b').zfill(4),
            2))
    return b[::2]
